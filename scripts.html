Spring Module v2.1

Used in top FPS games for camera bob, viewmodel sway, and smooth UI. Zero snap.

local Spring = {}
Spring.__index = Spring
function Spring.new(mass, force, damping, speed)
    return setmetatable({
        Target   = Vector3.new(), 
        Position = Vector3.new(),
        Velocity = Vector3.new(),
        Mass=mass or 1, Force=force or 50,
        Damping=damping or 4, Speed=speed or 4
    }, Spring)
end
function Spring:Update(dt)
    local d = self.Target - self.Position
    local f = d * self.Force - self.Velocity * self.Damping
    self.Velocity = self.Velocity + f / self.Mass * dt * self.Speed
    self.Position = self.Position + self.Velocity * dt * self.Speed
    return self.Position
end
return Spring
Resource Cleanup
✓ VERIFIED

Maid Class

The standard for memory-leak-free connections. Used in every professional Roblox codebase.

local Maid = {}
Maid.__index = Maid
function Maid.new()
    return setmetatable({_tasks={}}, Maid)
end
function Maid:Give(task)
    table.insert(self._tasks, task)
    return task
end
function Maid:Clean()
    for _, t in ipairs(self._tasks) do
        if type(t)=="function" then t()
        elseif typeof(t)=="RBXScriptConnection" then t:Disconnect()
        elseif t and t.Destroy then t:Destroy()
        end
    end
    self._tasks = {}
end
return Maid

Signal Class (GoodSignal)

Based on the GoodSignal open-source library. Full API parity with RBXScriptSignal, zero memory leaks.

local Signal = {}
Signal.__index = Signal
function Signal.new()
    return setmetatable({_listeners={}}, Signal)
end
function Signal:Connect(fn)
    table.insert(self._listeners, fn)
    return {Disconnect=function()
        for i,v in ipairs(self._listeners) do
            if v==fn then table.remove(self._listeners,i) break end
        end
    end}
end
function Signal:Fire(...)
    for _, fn in ipairs(self._listeners) do
        task.spawn(fn, ...)
    end
end
return Signal

Team-Aware Targeting + Spring Smooth

The closest enemy target lock with team check, head targeting, and Spring smoothing. One target at a time. No snap/tweak.

-- Paste AFTER the Spring module above
local cam = workspace.CurrentCamera
local lp  = game.Players.LocalPlayer
local smooth = Spring.new(1, 14, 4, 1)

local function getTarget(maxDist)
    local best, bestDist = nil, maxDist or 500
    for _, p in ipairs(game.Players:GetPlayers()) do
        if p ~= lp and p.Team ~= lp.Team then
            local char = p.Character
            local head = char and char:FindFirstChild("Head")
            if head then
                local d = (head.Position - cam.CFrame.Position).Magnitude
                if d < bestDist then best=head; bestDist=d end
            end
        end
    end
    return best
end

game:GetService("RunService").RenderStepped:Connect(function(dt)
    local target = getTarget()
    if target and game:GetService("UserInputService")
        :IsMouseButtonPressed(Enum.UserInputType.MouseButton2) then
        smooth.Target = target.Position
        smooth:Update(dt)
        cam.CFrame = CFrame.new(cam.CFrame.Position, smooth.Position)
    end
end)

Autonomous Input Control

Uses VirtualInputManager for genuine keyboard event simulation in rhythm games.

-- OSU! Mania Auto-Player Research
local VIM = game:GetService("VirtualInputManager")
local Keys = {Enum.KeyCode.D, Enum.KeyCode.F, Enum.KeyCode.J, Enum.KeyCode.K}

local function Tap(lane)
    VIM:SendKeyEvent(true, Keys[lane], false, game)
    task.wait(0.05)
    VIM:SendKeyEvent(false, Keys[lane], false, game)
end

-- Connect to note spawn events
workspace.DescendantAdded:Connect(function(note)
    if note:IsA("BasePart") and note.Name:find("Note") then
        local lane = note:GetAttribute("Lane") or 1
        task.spawn(Tap, math.clamp(lane, 1, 4))
    end
end)
